substituir ocorrencias em arquivos.ts

/* exemplo para um arquivo */

(Get-Content 'src/data/home/homeData.ts') -replace 'Lorem ipsum, dolor sit amet consectetur adipisicing', 'Lorem ipsum sit amet consec' |
Set-Content 'src/data/home/homeData.ts'

/* para varios arquivos */

Get-ChildItem -Path . -Filter *.tsx -Recurse |
ForEach-Object {
    (Get-Content $_.FullName) -replace 'HomeDataContext', 'HomeContext' |
    Set-Content $_.FullName
}

/* Arquivos de extensões diferentes */

Get-ChildItem -Path . -Recurse |
Where-Object { $_.Extension -match '^\.(ts|tsx)$' } |
ForEach-Object {
    (Get-Content $_.FullName) -replace 'soft-shadow', 'shadow-soft' |
    Set-Content $_.FullName
}

/* trocando variaveis */

# Cria a string de busca escapada: '\$\{\inputClasses\}'
$SearchPattern = [regex]::Escape('${inputClasses}')

Get-ChildItem -Path . -Filter *.tsx -Recurse |
    ForEach-Object {
        # Usa a variável escapada para garantir a busca literal.
        (Get-Content $_.FullName) -replace $SearchPattern, 'input' |
        Set-Content $_.FullName
    }
